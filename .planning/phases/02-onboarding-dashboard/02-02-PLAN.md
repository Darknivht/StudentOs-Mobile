1. **src/hooks/useStreak.ts** — Fetches streak data from Supabase profiles table per D-05:

   ```typescript
   import { useState, useEffect, useCallback } from "react";
   import { supabase } from "../services/supabase/client";
   import { useAuthStore } from "../stores/authStore";

   interface StreakData {
     currentStreak: number;
     longestStreak: number;
   }

   export function useStreak() {
     const user = useAuthStore((s) => s.user);
     const [data, setData] = useState<StreakData>({
       currentStreak: 0,
       longestStreak: 0,
     });
     const [isLoading, setIsLoading] = useState(true);
     const [error, setError] = useState<string | null>(null);

     const fetchStreak = useCallback(async () => {
       if (!supabase || !user) {
         setIsLoading(false);
         return;
       }
       setIsLoading(true);
       const { data: profile, error: err } = await supabase
         .from("profiles")
         .select("current_streak, longest_streak")
         .eq("id", user.id)
         .single();
       if (err) {
         setError(err.message);
       } else if (profile) {
         setData({
           currentStreak: profile.current_streak ?? 0,
           longestStreak: profile.longest_streak ?? 0,
         });
       }
       setIsLoading(false);
     }, [user]);

     useEffect(() => {
       fetchStreak();
     }, [fetchStreak]);
     return { data, isLoading, error, refetch: fetchStreak };
   }
   ```

2. **src/hooks/useStudyTime.ts** — Fetches today's study minutes and last 7 days trend per D-07:
   - Query `study_sessions` table where `user_id = user.id` and `created_at >= startOfToday`
   - Sum `duration_minutes` for today's total
   - Query last 7 days: group by date, sum duration_minutes per day
   - Return `{ todayMinutes: number, dailyTarget: number, weeklyData: Array<{date: string, minutes: number}> }`
   - Get `dailyTarget` from `useAppStore((s) => s.dailyStudyTarget)` (default 60)
   - Helper: `getStartOfToday()` returns new Date set to 00:00:00
   - Helper: `getLast7Days()` returns array of 7 date strings (today - 6 through today)
   - For weekly data, if a day has no sessions, return 0 minutes for that day
   - If supabase is null, return zeros and empty weeklyData

3. **src/hooks/useStudyProgress.ts** — Fetches today's activity counts per D-08:
   - Query 4 Supabase tables for today's counts (created_at >= startOfToday, user_id = user.id):
     - `notes`: count of notes created today
     - `quiz_attempts`: count of quizzes completed today
     - `flashcard_reviews`: count of flashcards reviewed today (use `flashcards` table with `updated_at >= startOfToday` as a proxy since flashcard_reviews table may not exist yet — use a try/catch fallback)
     - `focus_sessions`: sum of duration_minutes from focus_sessions today (or `study_sessions` as fallback)
   - Return `{ notesCreated: number, quizzesCompleted: number, flashcardsReviewed: number, focusMinutes: number }`
   - All 4 queries run in parallel via `Promise.all`
   - If any query fails, return 0 for that field (don't fail the whole hook)

4. **src/hooks/useAnnouncements.ts** — Fetches active announcements per D-09:
   - Query `announcements` table where `is_active = true` and `starts_at <= NOW()` and `(ends_at IS NULL OR ends_at >= NOW())`
   - Order by `created_at desc`
   - RLS handles tier/region filtering automatically (Supabase RLS policies on announcements table)
   - Return `{ announcements: Array<{id: string, title: string, message: string, type: 'info' | 'warning' | 'success', created_at: string}> }`
   - Limit to 3 most recent active announcements
   - If supabase is null or query fails, return empty array
