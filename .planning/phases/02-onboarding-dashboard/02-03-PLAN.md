1. **src/hooks/useBrainBoost.ts** — Brain Boost data and state management:

   ```typescript
   import { useState, useEffect, useCallback } from "react";
   import { supabase } from "../services/supabase/client";
   import { useAuthStore } from "../stores/authStore";

   interface BrainBoostQuestion {
     id: string;
     question: string;
     options: string[];
     correct_index: number;
   }

   interface BrainBoostState {
     questions: BrainBoostQuestion[];
     currentQuestionIndex: number;
     answers: number[]; // -1 = unanswered
     score: number;
     isCompleted: boolean;
     isCompletedToday: boolean;
     xpEarned: number;
     isLoading: boolean;
     error: string | null;
   }

   export function useBrainBoost() {
     const user = useAuthStore((s) => s.user);
     const [state, setState] = useState<BrainBoostState>({
       questions: [],
       currentQuestionIndex: 0,
       answers: [],
       score: 0,
       isCompleted: false,
       isCompletedToday: false,
       xpEarned: 0,
       isLoading: true,
       error: null,
     });

     // Check if already completed today
     const checkCompletion = useCallback(async () => {
       if (!supabase || !user) {
         setState((s) => ({ ...s, isLoading: false }));
         return;
       }
       const today = new Date().toISOString().split("T")[0]; // YYYY-MM-DD
       const { data: profile } = await supabase
         .from("profiles")
         .select("brain_boost_completed_at")
         .eq("id", user.id)
         .single();
       if (profile?.brain_boost_completed_at) {
         const completedDate = new Date(profile.brain_boost_completed_at)
           .toISOString()
           .split("T")[0];
         if (completedDate === today) {
           setState((s) => ({
             ...s,
             isCompletedToday: true,
             isCompleted: true,
             isLoading: false,
           }));
           return;
         }
       }
       setState((s) => ({ ...s, isLoading: false }));
     }, [user]);

     // Fetch 5 random questions
     const fetchQuestions = useCallback(async () => {
       if (!supabase || !user) return;
       setState((s) => ({ ...s, isLoading: true }));
       const { data, error } = await supabase
         .from("brain_boost_questions")
         .select("id, question, options, correct_index")
         .eq("is_active", true)
         .limit(5);
       // If no dedicated brain_boost_questions table, fall back to quiz_questions
       if (error || !data || data.length === 0) {
         const { data: quizData } = await supabase
           .from("quiz_questions")
           .select("id, question, options, correct_index")
           .limit(5);
         if (quizData && quizData.length > 0) {
           setState((s) => ({
             ...s,
             questions: quizData.slice(0, 5) as BrainBoostQuestion[],
             answers: new Array(Math.min(5, quizData.length)).fill(-1),
             isLoading: false,
           }));
           return;
         }
       }
       const questions = (data || []).slice(0, 5) as BrainBoostQuestion[];
       setState((s) => ({
         ...s,
         questions,
         answers: new Array(questions.length).fill(-1),
         isLoading: false,
       }));
     }, [user]);

     // Answer a question
     const answerQuestion = useCallback(
       (questionIndex: number, answerIndex: number) => {
         setState((prev) => {
           const newAnswers = [...prev.answers];
           newAnswers[questionIndex] = answerIndex;
           const isCorrect =
             prev.questions[questionIndex]?.correct_index === answerIndex;
           return {
             ...prev,
             answers: newAnswers,
             score: prev.score + (isCorrect ? 1 : 0),
           };
         });
       },
       [],
     );

     // Move to next question
     const nextQuestion = useCallback(() => {
       setState((prev) => ({
         ...prev,
         currentQuestionIndex: Math.min(
           prev.currentQuestionIndex + 1,
           prev.questions.length - 1,
         ),
       }));
     }, []);

     // Complete the boost and save to profile
     const completeBoost = useCallback(async () => {
       if (!supabase || !user) return;
       const xp = state.score * 10; // 10 XP per correct answer
       setState((prev) => ({
         ...prev,
         isCompleted: true,
         xpEarned: xp,
         isCompletedToday: true,
       }));
       await supabase
         .from("profiles")
         .update({
           brain_boost_completed_at: new Date().toISOString(),
           total_xp: supabase.rpc ? undefined : 0, // XP update handled by trigger ideally
         })
         .eq("id", user.id);
       // Also add XP via weekly_xp table if it exists
       await supabase.from("weekly_xp").upsert(
         {
           user_id: user.id,
           xp_earned: xp,
           source: "brain_boost",
           created_at: new Date().toISOString(),
         },
         { onConflict: "user_id,created_at,source" },
       );
     }, [user, state.score]);

     // Calculate time until midnight for countdown
     const getTimeUntilMidnight = useCallback(() => {
       const now = new Date();
       const midnight = new Date(now);
       midnight.setHours(24, 0, 0, 0);
       return midnight.getTime() - now.getTime(); // ms until midnight
     }, []);

     useEffect(() => {
       checkCompletion();
     }, [checkCompletion]);

     return {
       ...state,
       fetchQuestions,
       answerQuestion,
       nextQuestion,
       completeBoost,
       getTimeUntilMidnight,
     };
   }
   ```

1. **src/hooks/useCourses.ts** — Course list with progress calculation per D-11:
   - Local-first: query WatermelonDB for courses, then Supabase as fallback
   - Progress calculation per D-11: Notes 30%, Quizzes 30%, Flashcards 40%

   ```typescript
   import { useState, useEffect, useCallback } from "react";
   import { database, Course, Note, Flashcard } from "../database";
   import { supabase } from "../services/supabase/client";
   import { useAuthStore } from "../stores/authStore";

   interface CourseWithProgress {
     id: string;
     title: string;
     description?: string;
     color?: string;
     emoji?: string;
     notesProgress: number; // 0-100
     quizzesProgress: number; // 0-100
     flashcardsProgress: number; // 0-100
     overallProgress: number; // weighted 0-100
   }

   export function useCourses() {
     const user = useAuthStore((s) => s.user);
     const [courses, setCourses] = useState<CourseWithProgress[]>([]);
     const [isLoading, setIsLoading] = useState(true);

     const fetchCourses = useCallback(async () => {
       setIsLoading(true);
       try {
         // Local-first: query WatermelonDB
         const localCourses = await database
           .get<Course>("courses")
           .query()
           .fetch();

         if (localCourses.length > 0) {
           // Calculate progress for each course from local DB
           const coursesWithProgress = await Promise.all(
             localCourses.map(async (course) => {
               const courseId = course.id;
               // Count notes for this course
               const notesCount = await database
                 .get<Note>("notes")
                 .query(Q.where("course_id", courseId))
                 .fetchCount();
               // Count flashcards for this course
               const flashcardsCount = await database
                 .get<Flashcard>("flashcards")
                 .query(Q.where("course_id", courseId))
                 .fetchCount();

               // For progress, we need a "total expected" — use a baseline:
               // 10 notes = 100%, 5 quizzes = 100%, 20 flashcards = 100%
               // These are soft targets; progress caps at 100%
               const notesProgress = Math.min((notesCount / 10) * 100, 100);
               const quizzesProgress = 0; // No quiz_attempts table in WatermelonDB yet
               const flashcardsProgress = Math.min(
                 (flashcardsCount / 20) * 100,
                 100,
               );
               const overallProgress = Math.round(
                 notesProgress * 0.3 +
                   quizzesProgress * 0.3 +
                   flashcardsProgress * 0.4,
               );

               return {
                 id: course.id,
                 title: course.title,
                 description: course.description ?? undefined,
                 color: course.color ?? undefined,
                 emoji: course.emoji ?? undefined,
                 notesProgress,
                 quizzesProgress,
                 flashcardsProgress,
                 overallProgress,
               };
             }),
           );
           setCourses(coursesWithProgress);
         } else if (supabase && user) {
           // Fallback: fetch from Supabase
           const { data, error } = await supabase
             .from("courses")
             .select("id, title, description, color, emoji")
             .eq("user_id", user.id)
             .order("created_at", { ascending: false });

           if (!error && data) {
             const coursesWithProgress = data.map((c) => ({
               id: c.id,
               title: c.title,
               description: c.description ?? undefined,
               color: c.color ?? undefined,
               emoji: c.emoji ?? undefined,
               notesProgress: 0,
               quizzesProgress: 0,
               flashcardsProgress: 0,
               overallProgress: 0,
             }));
             setCourses(coursesWithProgress);
           }
         }
       } catch (err) {
         console.error("Failed to fetch courses:", err);
       }
       setIsLoading(false);
     }, [user]);

     useEffect(() => {
       fetchCourses();
     }, [fetchCourses]);
     return { courses, isLoading, refetch: fetchCourses };
   }
   ```

   - Import `Q` from `@nozbe/watermelondb` for queries
   - The "total expected" baselines (10 notes, 5 quizzes, 20 flashcards = 100%) are reasonable defaults that can be adjusted per course later. Progress caps at 100%.
